In [2]:
data = {
    'Product_ID': np.arange(101, 111),
    'Product_Name': ['A', 'B', 'C', 'D', 'E', 'F', 'G', 'H', 'I', 'J'],
    'Category': ['Electronics', 'Electronics', 'Furniture', 'Furniture', 'Clothing', 'Clothing', 'Food', 'Food', 'Books', 'Books'],
    'Price': np.round(np.random.uniform(10, 500, 10), 2),
    'Stock': np.random.randint(20, 200, 10)
}

Scatter plot¶

In [4]:
plt.figure(figsize=(10, 5))
plt.scatter(df['Price'], df['Stock'], color='blue', alpha=0.6)
plt.xlabel('Price')
plt.ylabel('Stock')
plt.title('Price vs Stock')
plt.grid(True)
plt.show()

Bar chart¶

In [5]:

plt.bar(df['Product_Name'], df['Stock'], color='green', alpha=0.7)
plt.xlabel('Product Name')
plt.ylabel('Stock')
plt.title('Stock per Product')

plt.show()
